3.테이블 만들기 

CREATE TABLE MEMBER ( 
	USER_NUM NUMBER PRIMARY KEY,
	ID VARCHAR2(30)  NOT NULL UNIQUE, 
	PASS VARCHAR2(30) NOT NULL,  
	EMAIL VARCHAR2(30) NOT NULL, 
	NICKNAME VARCHAR2(30) NOT NULL UNIQUE, 
	SIGNDATE DATE DEFAULT SYSDATE NOT NULL, 
	ADDRESS VARCHAR2(40) NOT NULL 
);

CREATE TABLE REVIEW ( 
	REVIEW_NUM NUMBER PRIMARY KEY, 
	TITLE VARCHAR2(30) NOT NULL, 
	REVIEW_CONTENT VARCHAR2(4000) NOT NULL, 
	ID VARCHAR2(30) NOT NULL UNIQUE, 
	NICKNAME VARCHAR2(30) NOT NULL UNIQUE, 
	RECOMMANDED NUMBER DEFAULT 0, 
	VISIT_COUNT NUMBER DEFAULT 0, 
	PERMISSION NUMBER CHECK ( PERMISSION IN ( 0, 1) ) ,
	CONSTRAINT ID_CONST FOREIGN KEY (ID) 
	REFERENCES MEMBER (ID),
	CONSTRAINT NICK_CONST FOREIGN KEY(NICKNAME) 
	REFERENCES MEMBER (NICKNAME)
);

CREATE TABLE COMMENT ( 
	COMMENT_NUM NUMBER PRIMARY KEY, 
	ID VARCHAR2(30) NOT NULL UNIQUE, 
	NICKNAME VARCHAR2(30) NOT NULL UNIQUE, 
	REVIEW_NUM NUMBER NOT NULL, 
	WRITEDATE DATE DEFAULT SYSDATE NOT NULL, 
	RECOMMANDED NUMBER DEFAULT 0, 
	CONSTRAINT ID_CONST FOREIGN KEY (ID) 
	REFERENCES MEMBER (ID),
	CONSTRAINT NICK_CONST FOREIGN KEY(NICKNAME) 
	REFERENCES MEMBER (NICKNAME), 
	CONSTRAINT REVIEW_CONST FOREIGN KEY(REVIEW_NUM) 
	REFERENCES REVIEW (REVIEW_NUM) 
);

CREATE TABLE BOARD (
	BOARD_NUM NUMBER PRIMARY KEY, 
	TITLE VARCHAR2(30) NOT NULL,
	BOARD_CONTENT VARCHAR2(4000) NOT NULL,
	ID VARCHAR2(30) NOT NULL UNIQUE, 
	NICKNAME VARCHAR2(30) NOT NULL UNIQUE, 
	WRITEDATE DATE DEFAULT SYSDATE, 
	RECOMMANDED NUMBER DEFAULT 0 ,
 	VISIT_COUNT NUMBER DEFAULT 0, 
	CONSTRAINT ID_CONST FOREIGN KEY (ID) 
	REFERENCES MEMBER (ID), 
	CONSTRAINT NICK_CONST FOREIGN KEY(NICKNAME) 
	REFERENCES MEMBER (NICKNAME) 
);

4. 시퀀스 생성

- 회원번호 증가 시퀀스 
CREATE SEQUENCE_USER_NUM 
	MINVALUE 1
	NOMAXVALUE 
	INCREMENT BY 1
	START WITH 1
	NOCACHE
	NOORDER
	NOCYCLE;

	NOORDER
